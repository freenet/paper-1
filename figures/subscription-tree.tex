% Subscription tree + UPDATE propagation. Every node in the tree is a
% peer that hosts the contract; "host" and "subscriber" are synonymous at
% the peer level (subscribe.rs:430-496: an end-subscriber fetches the
% contract body and announces hosting). UPDATE broadcasts may also pass
% through non-hosting forwarders along the path of a prior SUBSCRIBE;
% that per-contract forwarding role is described in the caption, not
% drawn here.
\begin{figure}[t]
\centering
\begin{tikzpicture}[
    host/.style={circle, draw=black!75, fill=orange!28, minimum size=9mm,
                 font=\scriptsize, inner sep=0pt, align=center},
    edge/.style={black!55, thick},
    update1/.style={->, >=Stealth, red!75!black, very thick},
    update2/.style={->, >=Stealth, blue!70!black, very thick, dashed},
    label/.style={font=\scriptsize, gray!35!black},
    >=Stealth,
]
  % Root host (close to l(C)).
  \node[host] (root) at (0,0) {host};

  % Two more hosting peers nearby.
  \node[host] (h1) at (-2.0,-1.4) {host};
  \node[host] (h2) at ( 2.0,-1.4) {host};

  % Mid-tier hosts (end-subscribers).
  \node[host] (a) at (-3.4,-2.8) {host};
  \node[host] (b) at (-1.6,-2.8) {host};
  \node[host] (c) at ( 0.5,-2.8) {host};
  \node[host] (d) at ( 2.6,-2.8) {host};
  \node[host] (e) at ( 4.2,-2.8) {host};

  \node[host] (a1) at (-4.3,-4.1) {host};
  \node[host] (b1) at (-1.6,-4.1) {host};
  \node[host] (e1) at ( 4.7,-4.1) {host};

  % Tree edges.
  \draw[edge] (root) -- (h1);
  \draw[edge] (root) -- (h2);
  \draw[edge] (h1) -- (a);
  \draw[edge] (h1) -- (b);
  \draw[edge] (h1) -- (c);
  \draw[edge] (h2) -- (d);
  \draw[edge] (h2) -- (e);
  \draw[edge] (a) -- (a1);
  \draw[edge] (b) -- (b1);
  \draw[edge] (e) -- (e1);

  % Update u1 injected at host a1, propagates up through a -> h1 -> root
  % and out through h1's other branches.
  \draw[update1] (a1.north east) to[bend left=12] node[label, sloped, above, text=red!75!black] {$u_1$} (a.south);
  \draw[update1] (a.north east) to[bend left=12] (h1.south west);
  \draw[update1] (h1.north) to[bend right=10] (root.south west);
  \draw[update1] (h1.south) to[bend left=10] (b.north);
  \draw[update1] (b.south) to[bend left=10] (b1.north);
  \draw[update1] (h1.south east) to[bend left=10] (c.north west);

  % Update u2 injected at host e1, propagates up through e -> h2 -> root.
  \draw[update2] (e1.north west) to[bend right=12] node[label, sloped, above, text=blue!70!black] {$u_2$} (e.south);
  \draw[update2] (e.north west) to[bend right=12] (h2.south east);
  \draw[update2] (h2.north) to[bend left=10] (root.south east);

  % Legend (top-right corner of the diagram).
  \begin{scope}[shift={(3.5,0.5)}]
    \draw[update1, line width=1pt] (-0.2,0) -- (0.25,0)
         node[font=\scriptsize, text=red!75!black, anchor=west, xshift=2pt]
         {$u_1$ injected lower-left};
    \draw[update2, line width=1pt] (-0.2,-0.4) -- (0.25,-0.4)
         node[font=\scriptsize, text=blue!70!black, anchor=west, xshift=2pt]
         {$u_2$ injected lower-right};
  \end{scope}

  % Annotation
  \node[font=\small, gray!35!black, anchor=west, align=left]
       at (-5.6,0.5) {hosting peers\\(near $\ell(\mathcal{C})$)};
  \draw[gray!50!black, ->, >=Stealth, thin] (-3.3,0.2) -- (root.west);

\end{tikzpicture}
\caption{Subscription tree for a contract $\mathcal{C}$. Every node
shown is a \emph{host} of $\mathcal{C}$: a peer that stores its
state and is also subscribed (an end-subscriber automatically
becomes a host, since the subscription requires fetching the
contract body). An \textsf{UPDATE} injected at any host is merged
locally and then propagated outward along tree edges using the
summary/delta protocol. Two concurrent updates $u_1$ and $u_2$
injected at opposite ends travel inward, meet at internal hosts,
and merge by $\mathcal{C}$'s commutative merge; the order in which
they arrive does not change the final state. The implementation
also forwards update broadcasts through some non-hosting peers
along the path of a prior \textsf{SUBSCRIBE} request, so the actual
fan-out graph may include additional forwarding nodes that the
figure omits for clarity.}
\label{fig:subscription-tree}
\end{figure}
